Champagne Tower Simulation Problem

def champagne(poured, row, glass):
    dp = [[0]*101 for _ in range(101)]
    dp[0][0] = poured

    for i in range(row+1):
        for j in range(i+1):
            if dp[i][j] > 1:
                extra = (dp[i][j] - 1) / 2
                dp[i][j] = 1
                dp[i+1][j] += extra
                dp[i+1][j+1] += extra

    return min(1, dp[row][glass])
